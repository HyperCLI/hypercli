## Main Client

```ts
import { HyperCLI } from "@hypercli.com/sdk";

const client = new HyperCLI({
  apiKey: "hyper_...",
  apiUrl: "https://api.hypercli.com",
  agentApiKey: "sk-...",
  agentsApiBaseUrl: "https://api.hypercli.com/agents",
  agentsWsUrl: "wss://api.agents.hypercli.com/ws",
  timeout: 30_000,
});
```
